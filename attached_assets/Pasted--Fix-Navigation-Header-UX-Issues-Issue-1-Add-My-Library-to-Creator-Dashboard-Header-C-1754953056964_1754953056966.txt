## Issue 3: Make Headers Consistent Across All Pages

### Create a Reusable Header Component
**Create `components/Header.tsx`:**

```javascript
// Universal header component that adapts to different pages
interface HeaderProps {
  variant?: 'default' | 'dashboard' | 'video';
  showBackButton?: boolean;
  title?: string;
}

const Header: React.FC<HeaderProps> = ({ 
  variant = 'default', 
  showBackButton = false, 
  title 
}) => {
  const navigate = useNavigate();
  const { user } = useAuth();
  
  return (
    <header className="bg-white shadow-sm border-b">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
        <div className="flex justify-between items-center h-16">
          {/* Left side */}
          <div className="flex items-center space-x-4">
            {showBackButton && (
              <>
                <button 
                  onClick={() => navigate(-1)}
                  className="flex items-center text-gray-600 hover:text-gray-900"
                >
                  <ArrowLeftIcon className="h-5 w-5 mr-1" />
                  Back
                </button>
                <div className="h-6 w-px bg-gray-300"></div>
              </>
            )}
            
            <Link to="/" className="flex items-center">
              <img src="/logo.svg" alt="Mosswood" className="h-8 w-8" />
              <span className="ml-2 text-xl font-semibold">Mosswood</span>
            </Link>
            
            {title && (
              <>
                <div className="h-6 w-px bg-gray-300"></div>
                <h1 className="text-lg font-medium text-gray-900">{title}</h1>
              </>
            )}
          </div>
          
          {/* Right side navigation */}
          <div className="flex items-center space-x-4">
            {variant === 'dashboard' && (
              <nav className="hidden md:flex space-x-6">
                <Link to="/dashboard" className="text-gray-600 hover:text-gray-900">
                  Dashboard
                </Link>
                <Link to="/dashboard/videos" className="text-gray-600 hover:text-gray-900">
                  Videos
                </Link>
              </nav>
            )}
            
            <Link 
              to="/explore"
              className="text-gray-600 hover:text-gray-900 px-3 py-2 rounded-md text-sm font-medium"
            >
              Explore
            </Link>
            
            <Link 
              to="/library"
              className="text-gray-600 hover:text-gray-900 px-3 py-2 rounded-md text-sm font-medium"
            >
              My Library
            </Link>
            
            {user ? (
              <UserMenu user={user} />
            ) : (
              <div className="flex space-x-2">
                <Link to="/login" className="text-gray-600 hover:text-gray-900">
                  Sign In
                </Link>
                <Link to="/signup" className="bg-blue-600 text-white px-4 py-2 rounded-md text-sm font-medium hover:bg-blue-700">
                  Sign Up
                </Link>
              </div>
            )}
          </div>
        </div>
      </div>
    </header>
  );
};
```
